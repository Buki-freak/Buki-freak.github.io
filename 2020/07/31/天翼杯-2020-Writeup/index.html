import gmpy2
from Crypto.Util.number import long_to_bytes

def find_q(d0, kbits, e, n, p):
    X = var('X')
    for k in range(1, e+1):
        results = solve_mod([e*d0*X - k*(p-1)*X*(n//p-X+1) + (p-1)*k*(n//p) == X], 2^kbits)
        for x in results:
            q0 = ZZ(x[0])
            if n % q0 == 0:
                return q0
